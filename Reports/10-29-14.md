  def leaf(m,one,node):
  if node._kids:
    east = node.east
    west = node.west
    mid_cos = node.mid_cos
    a = dist(m,one,west)
    b = dist(m,one,east)
    c = dist(m,west,east)
    x = (a*a + c*c - b*b)/(2*c)
    if (x<mid_cos):
      return leaf(m,one,node._kids[0])
    else:
      return leaf(m,one,node._kids[1])
  return node
